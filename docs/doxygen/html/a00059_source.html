92 /**
93  * Reads an unpacked GTFS feed
94  * from the given directory to the given database.
95  *
96  * This function reads every feed file and stores its records
97  * directly into the database, without an intermediate array.
98  *
99  * Unless CGTFS_STORING_BATCH_TRANSACTIONS_OFF is defined, each
100  * file's contents are stored in a single transaction.
101  *
102  * Pass NULL to `feed_counter` when there's no need
103  * to get the count of stored records.
104  *
105  * @param[in] dir /path/to/unpacked/gtfs/
106  * @param[out] db Initialized feed database instance to store into.
107  * @param[out] feed_counter Initialized feed instance to write stored records counts to.
108  *
109  * @returns Both FEED_DB_SUCCESS and FEED_DB_PARTIAL are successful results;
110  * FEED_DB_SUCCESS is returned only if the feed has records in
111  * EVERY possible feed file.
112  * FEED_DB_PARTIAL is returned if at least one entity file
113  * had at least one record; if not, FEED_DB_ERROR is returned.
114  *
115  * @ingroup Database__FeedEntity
116  */
117 feed_db_status_t store_feed_db(const char *dir, feed_db_t *db, feed_t *feed_counter);
