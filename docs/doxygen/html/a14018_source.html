60 /**
61  * Reads an unpacked GTFS feed
62  * from the given directory to the given database.
63  *
64  * This function reads every feed file and stores its records
65  * directly into the database, without an intermediate array.
66  *
67  * Unless CGTFS_STORING_BATCH_TRANSACTIONS_OFF is defined, each
68  * file's contents are stored in a single transaction.
69  *
70  * Pass NULL to `feed_counter` when there's no need
71  * to get the count of stored records.
72  *
73  * @param[in] dir /path/to/unpacked/gtfs/
74  * @param[out] db Initialized feed database instance to store into.
75  * @param[out] feed_counter Initialized feed instance to write stored records counts to.
76  *
77  * @returns Both FEED_DB_SUCCESS and FEED_DB_PARTIAL are successful results;
78  * FEED_DB_SUCCESS is returned only if the feed has records in
79  * EVERY possible feed file.
80  * FEED_DB_PARTIAL is returned if at least one entity file
81  * had at least one record; if not, FEED_DB_ERROR is returned.
82  *
83  * @ingroup Database__FeedEntity
84  */
85 feed_db_status_t store_feed_db(const char *dir, feed_db_t *db, feed_t *feed_counter);
